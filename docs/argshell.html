245    def columnize(self, list_: list[str] | None, displaywidth: int = 80):  # type: ignore
246        """Display a list of strings as a compact set of columns.
247
248        Each column is only as wide as necessary.
249        Columns are separated by two spaces (one was not legible enough).
250        """
251        if not list_:
252            console.print(f"[bright_red]<empty>")
253            # self.stdout.write("<empty>\n")
254            return
255
256        nonstrings = [i for i in range(len(list_)) if not isinstance(list_[i], str)]  # type: ignore
257        if nonstrings:
258            raise TypeError(
259                "list[i] not a string for i in %s" % ", ".join(map(str, nonstrings))
260            )
261        size = len(list_)
262        if size == 1:
263            self.stdout.write("%s\n" % str(list_[0]))
264            console.print(f"[turquoise2]{list_[0]}")
265            return
266        # Try every row count from 1 upwards
267        for nrows in range(1, len(list_)):
268            ncols = (size + nrows - 1) // nrows
269            colwidths: list[int] = []
270            totwidth = -2
271            for col in range(ncols):
272                colwidth = 0
273                for row in range(nrows):
274                    i = row + nrows * col
275                    if i >= size:
276                        break
277                    x = list_[i]
278                    colwidth = max(colwidth, len(x))
279                colwidths.append(colwidth)
280                totwidth += colwidth + 2
281                if totwidth > displaywidth:
282                    break
283            if totwidth <= displaywidth:
284                break
285        else:
286            nrows = len(list_)
287            ncols = 1
288            colwidths = [0]
289        for row in range(nrows):
290            texts: list[str] = []
291            for col in range(ncols):
292                i = row + nrows * col
293                if i >= size:
294                    x = ""
295                else:
296                    x = list_[i]
297                texts.append(x)
298            while texts and not texts[-1]:
299                del texts[-1]
300            for col in range(len(texts)):
301                texts[col] = texts[col].ljust(colwidths[col])
302
303            console.print(f"[turquoise2]{'  '.join(texts)}")

421        @wraps(func)

423            try:
424                args = parser().parse_args(shlex.split(command))
425            except Exception as e:
426                # On parser error, print help and skip post_parser and func execution
427                if "the following arguments are required" not in str(e):

437                return None
438            for post_parser in post_parsers:
439                args = post_parser(args)
